edict_reduce_devastation = {
	potential = {
		always = yes		#we support "potential" if modders want to have lots and just show some.
	}
	
	allow = {
		always = yes
	}
	
	modifier = {
		local_monthly_devastation = -0.20
	}
	
	color = {  204 50 254  }
	
	
	ai_will_do = {
		factor = 100
		modifier = {
			factor = 0
			AND = {
				has_state_edict = edict_of_absolutism
				all_province_in_state = {
					NOT = {
						devastation = 1
					}	
				}
			}
		}
		modifier = {
			factor = 5
			has_state_edict = edict_of_absolutism
		}
		modifier = {
			factor = 0
			
			AND = {
				NOT = { has_state_edict = edict_of_absolutism }
				all_province_in_state = {
					NOT = {
						devastation = 2
					}
				}
			}
		}
		modifier = {
			factor = 5
			
			AND = {
				NOT = { has_state_edict = edict_of_absolutism }
				all_province_in_state = {
					devastation = 5
				}
			}
		}
		modifier = {
			factor = 5
			
			AND = {
				NOT = { has_state_edict = edict_of_absolutism }
				any_province_in_state = {
					devastation = 10
				}
			}
		}
	}
}
